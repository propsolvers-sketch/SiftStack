```python
# src/observability.py — public contracts

from collections import OrderedDict
from pathlib import Path
from typing import Iterable

STATE_FILE: Path  # = Path("output/observability/service_rates.json")

TRACKED_SERVICES: tuple[str, ...]  # = ("2captcha", "smarty", "tracerfy", "llm")


class FunnelCounter:
    """Ordered gate-count recorder for a single pipeline run."""

    def __init__(self, pipeline_name: str, gates: Iterable[str] | None = None) -> None:
        """If `gates` is given, pre-seed the ordered key list at 0 so the
        Slack block always shows every gate even if a stage emits zero."""

    def set(self, gate: str, count: int) -> None:
        """Idempotent record (last write wins). Preserves insertion order."""

    def increment(self, gate: str, by: int = 1) -> None:
        """Bump (or initialize) a gate's count."""

    def as_ordered_dict(self) -> OrderedDict[str, int]:
        """Return the gate sequence in insertion order for the Slack block."""

    @property
    def pipeline_name(self) -> str: ...


class ServiceRateTracker:
    """Tallies success/failure per call-site for the 4 tracked services.
    Unknown service names are silently ignored (no raise) so call sites
    can be wrapped defensively."""

    def __init__(self) -> None: ...

    def record(self, service: str, success: bool) -> None:
        """service must be one of TRACKED_SERVICES; unknown -> no-op."""

    def totals(self) -> dict[str, dict[str, int]]:
        """Returns {service: {"success": int, "total": int}} for tracked services only."""

    def per_run_rates(self) -> dict[str, float | None]:
        """Returns {service: success/total} for tracked services, None when total==0."""


def load_rolling_rates(state_file: Path = STATE_FILE) -> dict[str, list[dict]]:
    """Returns persisted rolling state: {service: [{date, success, total}, ...]}.
    Returns {} when file missing or corrupt (does not raise)."""


def save_rolling_rates(
    today_totals: dict[str, dict[str, int]],
    *,
    today_date: str | None = None,   # ISO YYYY-MM-DD; defaults to today
    state_file: Path = STATE_FILE,
    keep_days: int = 7,
) -> dict[str, list[dict]]:
    """Atomic write: read existing, replace/append today's entry per service,
    prune entries older than `keep_days`, write to tmp + rename. Creates
    the parent directory if missing. Returns the persisted dict for chaining."""


def rolling_rates_summary(
    rolling: dict[str, list[dict]],
) -> dict[str, float | None]:
    """Collapse rolling entries into a single {service: success/total} dict
    summing across all retained days. None when total==0."""
```

```python
# src/slack_notifier.py — appended helpers

def build_funnel_block(
    pipeline_name: str,
    gate_counts: dict[str, int] | "OrderedDict[str, int]",
) -> dict:
    """Returns a Slack Block Kit "section" dict rendering the per-pipeline
    funnel. Pipeline-specific gate sequence is honored (per D-01 — per-pipeline,
    not unified). Block shape:
        {
          "type": "section",
          "text": {"type": "mrkdwn", "text": "*Funnel — {pipeline_name}*\n• scraped: 47\n• zip-gated: 12\n..."}
        }
    Caller appends this dict to its existing blocks list before _send_webhook."""


def build_service_rates_block(
    per_run_rates: dict[str, float | None],
    rolling_rates: dict[str, float | None],
) -> dict:
    """Returns a Slack Block Kit "section" dict rendering 4-service rates as:
        Smarty: 86% today | 92% 7-day
        2Captcha: 100% today | 99% 7-day
        Tracerfy: 33% today | 41% 7-day
        LLM: 95% today | 96% 7-day
    Services with total==0 today show 'n/a today' rather than '0%'. Rolling
    'n/a' shows '— 7-day'. Order: 2Captcha, Smarty, Tracerfy, LLM."""
```
