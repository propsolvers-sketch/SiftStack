```python
from observability import (
    FunnelCounter, ServiceRateTracker,
    load_rolling_rates, save_rolling_rates, rolling_rates_summary,
)
from slack_notifier import (
    build_funnel_block, build_service_rates_block, _send_blocks_webhook,
)

# At orchestrator entry:
funnel = FunnelCounter("tax_distress", gates=[
    "bulk_fetched", "individual_owner_filtered", "min_balance_filtered",
    "smarty_geocoded", "tier_gated",
])
rate_tracker = ServiceRateTracker()

# After each stage: funnel.set("<gate>", surviving_count)

# notify_slack helper (NEW — neither pipeline has one today):
def notify_slack(notices, funnel, rate_tracker, *, webhook_url=None):
    """Post a single Slack message with per-run summary + funnel + service-rates blocks."""
    text = _build_summary_text(notices, funnel)  # local helper — short summary header

    # Rolling-rates ordering (WARNING 6 resolution):
    rolling = rolling_rates_summary(load_rolling_rates())  # BEFORE blocks build
    per_run = rate_tracker.per_run_rates()

    blocks = [
        {"type": "section", "text": {"type": "mrkdwn", "text": text}},
        build_funnel_block(funnel.pipeline_name, funnel.as_ordered_dict()),
        build_service_rates_block(per_run, rolling),
    ]
    sent = _send_blocks_webhook(text, blocks, webhook_url=webhook_url)
    if sent:
        save_rolling_rates(rate_tracker.totals())  # AFTER successful send
    return sent
```
